```python
def update_from_paper_trade(trade: dict) -> bool:
    """Paper-train-only safe learning update. Never calls lm_update. Never calls unsafe legacy feature iterators."""
    try:
        t = trade if isinstance(trade, dict) else {}
        sym = str(t.get("symbol") or "UNKNOWN")
        reg = str(t.get("regime") or "UNKNOWN")
        bucket = str(t.get("training_bucket") or t.get("explore_bucket") or t.get("bucket") or "UNKNOWN")
        pnl_pct = _safe_float(t.get("net_pnl_pct"), 0.0)
        pnl = _safe_float(t.get("pnl_decimal"), pnl_pct / 100.0)
        ws = _safe_float(t.get("score_at_entry", t.get("ws", 0.0)), 0.0)
        features = _safe_features(t.get("features"))

        # IMPORTANT:
        # This path must NOT call lm_update().
        # This path must NOT call record_features() unless record_features is scalar-safe and tested.

        key = f"{sym}_{reg}"

        # Use existing module globals if present; otherwise create safe dicts.
        # Keep this minimal and robust.
        global model_state
        if "model_state" not in globals() or not isinstance(model_state, dict):
            model_state = {}

        paper = model_state.setdefault("paper_train_learning", {})
        row = paper.setdefault(key, {
            "symbol": sym,
            "regime": reg,
            "n": 0,
            "wins": 0,
            "losses": 0,
            "flats": 0,
            "pnl_sum": 0.0,
            "pnl_history": [],
            "score_history": [],
            "bucket_counts": {},
            "feature_counts": {},
        })

        row["n"] = int(row.get("n") or 0) + 1
        outcome = str(t.get("outcome") or "").upper()
        if outcome == "WIN":
            row["wins"] = int(row.get("wins") or 0) + 1
        elif outcome == "LOSS":
            row["losses"] = int(row.get("losses") or 0) + 1
        else:
            row["flats"] = int(row.get("flats") or 0) + 1

        row["pnl_sum"] = float(row.get("pnl_sum") or 0.0) + pnl
        hist = row.setdefault("pnl_history", [])
        if isinstance(hist, list):
            hist.append(pnl)
            del hist[:-200]

        sh = row.setdefault("score_history", [])
        if isinstance(sh, list):
            sh.append(ws)
            del sh[:-200]

        bc = row.setdefault("bucket_counts", {})
        if isinstance(bc, dict):
            bc[bucket] = int(bc.get(bucket) or 0) + 1

        fc = row.setdefault("feature_counts", {})
        if isinstance(fc, dict):
            for name, value in features.items():
                # count feature presence only; never iterate scalar values
                fc[str(name)] = int(fc.get(str(name)) or 0) + 1

        # Optional: call existing persistence only if it is safe/no-arg and already used.
        safe_save_model_state_if_available()

        return True
    except Exception:
        log.exception("[PAPER_TRAIN_LEARNING_INTERNAL_ERROR] update_from_paper_trade failed")
        return False
```

Required helpers:
```python
def _safe_float(v, default=0.0):
    try:
        if v is None:
            return default
        return float(v)
    except Exception:
        return default

def _safe_features(raw):
    if not isinstance(raw, dict):
        return {}
    out = {}
    for k, v in raw.items():
        name = str(k)
        if isinstance(v, (int, float, bool, str)) or v is None:
            out[name] = v
        elif isinstance(v, dict):
            out[name] = dict(v)
        elif isinstance(v, (list, tuple, set)):
            out[name] = list(v)
        else:
            out[name] = str(v)
    return out
```

If `safe_save_model_state_if_available()` does not exist, implement as no-op wrapper around existing save:

def safe_save_model_state_if_available():
    try:
        fn = globals().get("save_model_state") or globals().get("_save_model_state")
        if callable(fn):
            fn()
    except TypeError:
        # If save requires args, skip; do not break learning update.
        return
    except Exception:
        log.exception("[PAPER_TRAIN_LEARNING_SAVE_ERROR]")
