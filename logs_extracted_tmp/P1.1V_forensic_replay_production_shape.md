cd /opt/cryptomaster
PYTHONPATH=. python3 - <<'PY'
import json, copy, traceback, pprint
from src.services.learning_monitor import update_from_paper_trade

# Load real positions from production
try:
    with open("data/paper_open_positions.json", "r") as f:
        positions_data = json.load(f)
except Exception as e:
    print(f"ERROR loading positions: {e}")
    exit(1)

print(f"[FORENSIC_START] Loaded {len(positions_data) if isinstance(positions_data, dict) else 0} positions")
print(f"Type of positions_data: {type(positions_data).__name__}")

if not positions_data:
    print("No positions to test")
    exit(0)

# Convert list to dict if needed (legacy format)
if isinstance(positions_data, list):
    positions_data = {str(i): p for i, p in enumerate(positions_data)}
    print(f"[FORENSIC] Converted list to dict: {len(positions_data)} items")

# Replay each position as if it's closing
for trade_id, pos in positions_data.items():
    t = copy.deepcopy(pos)
    
    # Simulate a close: add required fields if missing
    if "net_pnl_pct" not in t:
        t["net_pnl_pct"] = -0.1234
    if "outcome" not in t:
        t["outcome"] = "LOSS"
    if "exit_reason" not in t:
        t["exit_reason"] = "FORENSIC_TEST"
    if "reason" not in t:
        t["reason"] = "FORENSIC_TEST"
    
    # Ensure training_bucket is set for paper_train
    if "training_bucket" not in t or not t.get("training_bucket"):
        t["training_bucket"] = t.get("explore_bucket") or "C_WEAK_EV_TRAIN"
    
    sym = t.get("symbol", "UNKNOWN")
    bucket = t.get("training_bucket", "UNKNOWN")
    
    print(f"\n[FORENSIC_REPLAY] trade_id={trade_id[:8]}... sym={sym} bucket={bucket}")
    print(f"  position keys: {list(t.keys())[:5]}... ({len(t)} total)")
    
    # Show field types that might cause iteration errors
    potentially_iterable_fields = {
        "features": t.get("features"),
        "tags": t.get("tags"),
        "scores": t.get("scores"),
        "pnl_history": t.get("pnl_history"),
    }
    
    print(f"  field types to check:")
    for fname, fval in potentially_iterable_fields.items():
        if fval is not None:
            ftype = type(fval).__name__
            print(f"    {fname}: {ftype}")
            # If dict, show inner types
            if isinstance(fval, dict) and len(fval) <= 5:
                inner_types = {k: type(v).__name__ for k, v in fval.items()}
                print(f"      inner: {inner_types}")
    
    # Now call the real function
    try:
        result = update_from_paper_trade(t)
        print(f"  ✓ PASSED: result={result}")
    except Exception as e:
        print(f"  ✗ FAILED: {type(e).__name__}: {e}")
        print(f"\n[FORENSIC_TRACEBACK] Full trace:")
        traceback.print_exc()
        print(f"\n[FORENSIC_POSITION_DUMP] Failed position:")
        pprint.pprint(t)
        exit(1)

print(f"\n[FORENSIC_END] All {len(positions_data)} positions replayed successfully")
exit(0)

**Fix:**
- Ensure dedup is working: add log at line 866 to confirm it's skipping duplicates
- OR ensure trade_executor.py doesn't process paper_train trades
- OR guard update_bucket_metrics in trade_executor to skip paper_train
